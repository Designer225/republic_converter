
convert_republic_government = {
	switch = {
		trigger = has_game_rule
		republic_conversion_default = {
			if = {
				limit = {
					is_independent_ruler = no
					liege = { government_allows = administrative }
				}
				switch = {
					trigger = has_government
					celestial_government = {
						convert_republic_switch_to_celestial = yes
					}
					meritocratic_government = {
						convert_republic_switch_to_meritocratic = yes
					}
					steppe_admin_government = {
						convert_republic_switch_to_steppe_admin = yes
					}
					japan_administrative_government = {
						convert_republic_switch_to_japan_administrative = yes
					}
					fallback = {
						convert_republic_switch_to_administrative = yes
					}
				}
			}
			else_if = {
				limit = {
					location = {	
						OR = {
							geographical_region = world_asia_china
							geographical_region = custom_dynastic_cycle
						}
					}
				}
				convert_republic_switch_to_celestial = yes
			}
			else_if = {
				limit = {
					location = { geographical_region = world_steppe }
					OR = {
						AND = {
							is_independent_ruler = no
							liege = { is_nomad = yes }
						}
						any_neighboring_top_liege_realm_owner = { is_nomad = yes }
					}
				}
				convert_republic_switch_to_steppe_admin = yes
			}
			else_if = {
				limit = {
					location = { geographical_region = world_asia_japan }
					NOT = { has_global_variable = shogunate_established }
				}
				convert_republic_switch_to_japan_administrative = yes
			}
			else_if = {
				limit = {
					OR = {
						location = { geographical_region = world_asia } # this seems to refer to East Asia as a whole, including Northeast and Southeast
						AND = {
							location = { geographical_region = world_asia_japan }
							has_global_variable = shogunate_established # shogunate abolished ritsuryo
						}
					}
				}
				convert_republic_switch_to_meritocratic = yes
			}
			else = {
				convert_republic_switch_to_administrative = yes
			}
		}
		republic_conversion_administrative = {
			convert_republic_switch_to_administrative = yes
		}
		republic_conversion_meritocratic = {
			convert_republic_switch_to_meritocratic = yes
		}
		republic_conversion_steppe_admin = {
			convert_republic_switch_to_steppe_admin = yes
		}
		republic_conversion_celestial = {
			convert_republic_switch_to_celestial = yes
		}
		republic_conversion_ritsuryo = {
			convert_republic_switch_to_japan_administrative = yes
		}
	}
}

convert_republic_switch_to_administrative = {
	if = {
		limit = { has_ep3_dlc_trigger = yes }
		change_government = administrative_government
		convert_republic_switch_gov_fallback_effect = yes
		if = {
			limit = { government_allows = administrative }
			give_new_noble_family_title_effect = yes
		}
	}
}

convert_republic_switch_gov_fallback_effect = {
	switch = {
		trigger = has_game_rule
		republic_conversion_under_main_tier_handling_title = {
			if = {
				limit = { highest_held_title_tier < main_administrative_tier }
				convert_republic_switch_gov_titular_fallback_effect = yes
			}
		}
		republic_conversion_under_main_tier_handling_none = {
			change_government = republic_government # revert conversion
		}
	}
}

convert_republic_switch_gov_titular_fallback_effect = {
	primary_title = {
		save_scope_as = old_title
	}
	if = {
		limit = { main_administrative_tier = tier_kingdom }
		create_dynamic_title = {
			tier = kingdom
			name = NEW_CREATED_TITLE_NAME
		}
	}
	else = {
		create_dynamic_title = {
			tier = duchy
			name = NEW_CREATED_TITLE_NAME
		}
	}
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	scope:new_title = {
		set_coa = scope:old_title
		set_color_from_title = scope:old_title
		set_delete_on_destroy = yes
		set_destroy_on_gain_same_tier = yes
		change_title_holder = {
			holder = root
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change
}
